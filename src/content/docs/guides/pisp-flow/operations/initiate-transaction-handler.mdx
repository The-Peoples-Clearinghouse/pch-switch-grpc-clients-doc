    async function initiateTransactionHandler(
      request: ServerInitiateTransactionRequest,
    ): Promise<ServerInitiateTransactionResponse> {
      const partyId          = request.getPartyid();
      const sourceFspId      = request.getSourcefspid();
      const destinationFspId = request.getDestinationfspid();
      const transactionId    = request.getTransactionid();
      const payeeId          = request.getPayeeid();
      const payeeFsp         = request.getPayeefsp();
      const accountId        = request.getAccountid();
      const consentToken     = request.getConsenttoken();
      const currencyCode     = request.getCurrencycode();
      const amount           = request.getAmount();
      const note             = request.getNote();
      const threeppienabled  = request.getThreeppienabled();
      const transactionType  = request.getTransactiontype();

      const response = new ServerInitiateTransactionResponse();
      // La fuente de la respuesta es el banco receptor (el nuestro)
      response.setSourcefspid(destinationFspId);
      // El destino de la respuesta es quien originÃ³ la solicitud
      response.setDestinationfspid(sourceFspId);
      response.setPartyid(partyId);
      response.setTransactionid(transactionId);

      // Verifica el consent token y que la cuenta estÃ© autorizada
      if (!consentTokenAutorizaCuenta(consentToken, accountId)) {
        const error = new ErrorResponse();
        error.setCode('UNAUTHORIZED');
        error.setMessage('El consent token no autoriza operaciones sobre la cuenta indicada.');
        response.setError(error);
        return response;
      }

      // Ejecuta la transferencia hacia el Switch mediante ExecuteTransfer
      // Ver: /guides/basic-flow/execute-transfer/
      const resultado = await ejecutarTransferencia({
        transactionId,
        payeeId,
        payeeFsp,
        accountId,
        currencyCode,
        amount,
        note,
        transactionType,
      });

      if (!resultado.exitoso) {
        const error = new ErrorResponse();
        error.setCode('TRANSFER_FAILED');
        error.setMessage('No fue posible ejecutar la transferencia.');
        response.setError(error);
        return response;
      }

      response.setTransactionstate(resultado.estado);

      return response;
    }

    // Registrar el handler en el cliente gRPC
    grpcClient.setInitiateTransactionRequestHandlerFn(initiateTransactionHandler);
